# To Do: Create a visualization that will support your conclusions for part two
# Make df for visualization
population_df = df.copy(deep=True)

# drop nulls
population_df = population_df[['Agency', 'Service Area Population', 'Unlinked Passenger Trips FY']].dropna()

# Prepare columns
population_df['Service Area Population'] = pd.to_numeric(
    population_df['Service Area Population'], errors = 'coerce')

population_df['Unlinked Passenger Trips FY'] = pd.to_numeric(
    population_df['Unlinked Passenger Trips FY'], errors = 'coerce')

# group data to prepare for visualization
population_df = population_df.groupby('Agency', as_index=False).agg({
    'Service Area Population' : 'first',
    'Unlinked Passenger Trips FY' : 'sum'
})

# fix scales
population_df['Population (Millions)'] = population_df['Service Area Population'] / 1_000_000
population_df['Trips'] = population_df['Unlinked Passenger Trips FY']

# init plot
fig, ax = plt.subplots(figsize=(8,5))

# remove 0s
population_df = population_df[population_df['Unlinked Passenger Trips FY'] > 0]

# plot data
ax.scatter(
    population_df['Population (Millions)'],
    population_df['Trips']
)

# set labels
ax.set_xlabel('Population (Millions)')
ax.set_ylabel('Unlinked Passenger Trips')
ax.set_title('Ridership vs. Population by Agency')

# stylize
ax.grid(True, linestyle = '--', alpha = 0.5)
